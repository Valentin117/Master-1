import smtplib
# Set up la connexion avec SMTP

smtp_port = 587

# Header du message
from_email = 'truc@gmail.com'
to_email = 'destinataire@gmail.com'
subject ='Anniversaire!'

message = f'{subject} {body}'

# Envoyer le message
smtp_connection.sendmail(from_email, to_email, message)
